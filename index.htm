Create and Play Your Own Games

1Initiate the framework UI

# Clone the repository
git clone https://github.com/RiddlerAI/riddler.git
cd riddler

# Install dependencies
npm install

# Set up environment variables (Set Up OPENAI_API_KEY)
cp .env.example .env

#Start the Framework UI
npm run play

2Manually generate and save a game

import { GameGenerator } from '../services/gameGeneratorService';
import { generateRandomTheme } from '../services/gameGeneratorService';

async function main() {
  try {
    # Generate a random theme
    const theme = await generateRandomTheme();
    console.log(`Generated theme: ${theme}`);

    # Create a new game
    const game = await GameGenerator.generateGame(theme, 'tutorial');
    console.log('Game generated successfully:', game);
  } catch (error) {
    console.error('Error generating game:', error);
  }
}

main();
